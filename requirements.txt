console = Console()

def banner():
    console.clear()
    f = Figlet(font='slant')
    ascii_art = f.renderText('HACKER AI')
    text = Text(ascii_art, style="bold cyan")
    panel = Panel(text, border_style="green", subtitle="v2.0 · Assistente Hacker")
    console.print(panel)

def get_available_models():
    avail = []
    if OPENROUTER_KEY and "sua-chave" not in OPENROUTER_KEY:
        avail += [m for m in MODELS if m["provider"] == "openrouter"]
    if MISTRAL_KEY and "sua-chave" not in MISTRAL_KEY:
        avail += [m for m in MODELS if m["provider"] == "mistral"]
    return avail

def model_menu(models):
    table = Table(title="🤖 Modelos Gratuitos Disponíveis", border_style="magenta", box=box.HEAVY)
    table.add_column("Nº", style="cyan", justify="right")
    table.add_column("Nome", style="white bold")
    table.add_column("Provedor", style="magenta")

    for i, m in enumerate(models, 1):
        prov = "OpenRouter" if m["provider"] == "openrouter" else "Mistral"
        prov_style = "magenta" if prov == "OpenRouter" else "cyan"
        table.add_row(str(i), m["name"], f"[{prov_style}]{prov}[/]")
    console.print(table)

    while True:
        try:
            sel = console.input("[green]╰─> Escolha o modelo: [/]")
            idx = int(sel) - 1
            if 0 <= idx < len(models):
                return models[idx]
            console.print("[red]Número inválido.[/]")
        except ValueError:
            console.print("[red]Digite um número.[/]")

def save_code(content, fname=None):
    if not fname:
        fname = console.input("[cyan]Nome do arquivo: [/]")
    if not fname:
        console.print("[red]Nome inválido.[/]")
        return
    try:
        with open(fname, "w", encoding="utf-8") as f:
            f.write(content)
        console.print(Panel(f"💾 Código salvo em {fname}", border_style="green"))
    except Exception as e:
        console.print(f"[red]Erro ao salvar: {e}[/]")

# 🚀 MAIN
# ═══════════════════════════════════════════════════════════════
def main():
    banner()
    models = get_available_models()
    if not models:
        console.print("[red]Nenhuma chave de API configurada. Edite o script.[/]")
        sys.exit(1)

    model = model_menu(models)

    if model["provider"] == "openrouter":
        api_key = OPENROUTER_KEY
        url = "https://openrouter.ai/api/v1/chat/completions"
        headers = {
            "Authorization": f"Bearer {api_key}",
            "Content-Type": "application/json"
        }
        if SITE_URL: headers["HTTP-Referer"] = SITE_URL
        if SITE_NAME: headers["X-Title"] = SITE_NAME
    else:
        api_key = MISTRAL_KEY
        url = "https://api.mistral.ai/v1/chat/completions"
        headers = {
            "Authorization": f"Bearer {api_key}",
            "Content-Type": "application/json"
        }

    system = {
        "role": "system",
        "content": (
            "Você é HACKER AI, um assistente hacker irônico e técnico. "
            "Responda sempre em português, com foco em programação, segurança e tecnologia. "
            "Use blocos de código com ``` quando for útil. Seja conciso mas explicativo."
        )
    }
    messages = [system]

    console.print(
        Panel(f"🔥 HACKER AI ATIVO  ·  {model['name']}",
              border_style="green",
              subtitle="/clear limpar | /quit sair")
    )

    while True:
        try:
            user_input = console.input("[bold white]▸ [/]")
        except (EOFError, KeyboardInterrupt):
            console.print("\n[green]Até mais, hacker![/]\n")
            break

        cmd = user_input.strip().lower()
        if cmd in ["/quit", "/exit", "sair"]:
            console.print("[green]Stay hacky! 👾[/]\n")
            break
        if cmd == "/clear":
            banner()
            console.print(
                Panel(f"🔥 HACKER AI  ·  {model['name']}",
                      border_style="green")
            )
            continue
        if not cmd:
            continue

        messages.append({"role": "user", "content": user_input})

        # Spinner enquanto espera a API
        with console.status("[yellow]⏳ Pensando...[/]", spinner="dots"):
            try:
                payload = {
                    "model": model["id"],
                    "messages": messages,
                    "temperature": 0.7,
                    "max_tokens": 1024
                }
                resp = requests.post(url, headers=headers, json=payload)
                resp.raise_for_status()
                reply = resp.json()["choices"][0]["message"]["content"]
            except Exception as e:
                console.print(f"[red]Erro na API: {e}[/]")
                messages.pop()
                continue

        # Exibe resposta com renderização Markdown (inclui código com syntax highlight nativo)
        console.print("🤖 ", end="")
        md = Markdown(reply, code_theme="monokai")
        console.print(md)

        messages.append({"role": "assistant", "content": reply})

        # Verifica blocos de código para download
        code_blocks = re.findall(r"```(?:\w+)?\n(.*?)```", reply, re.DOTALL)
        if code_blocks:
            console.print(f"\n[magenta]⚡ {len(code_blocks)} bloco(s) de código encontrado(s).[/]")
            save = console.input("[cyan]   Salvar? (y/n): [/]").strip().lower()
            if save == "y":
                if len(code_blocks) == 1:
                    save_code(code_blocks[0])
                else:
                    for i, code in enumerate(code_blocks, 1):
                        prev = code.split('\n')[0][:50]
                        console.print(f"   [[{i}]] {prev}...")
                    sel = console.input("[cyan]   Número ou 'todos': [/]").strip().lower()
                    if sel == "todos":
                        for i, code in enumerate(code_blocks, 1):
                            save_code(code, f"codigo_{i}.txt")
                    else:
                        try:
                            idx = int(sel) - 1
                            if 0 <= idx < len(code_blocks):
                                save_code(code_blocks[idx])
                        except:
                            console.print("[red]Inválido.[/]")
        console.print()

if __name__ == "__main__":
    main()
